-- Create buttons for each script

    Button.Size = UDim2.new(1, 0, 0, 40)
    Button.BackgroundColor3 = Color3.fromRGB(50, 50, 50)

    Button.TextColor3 = Color3.fromRGB(255, 255, 255)
    Button.TextSize = 16
    Button.Font = Enum.Font.Gotham
    Button.Parent = ButtonFrame
    local BtnCorner = Instance.new("UICorner")
    BtnCorner.CornerRadius = UDim.new(0, 5)
    BtnCorner.Parent = Button
    local BtnGradient = Instance.new("UIGradient")
    BtnGradient.Color = ColorSequence.new{
        ColorSequenceKeypoint.new(0, Color3.fromRGB(128, 0, 128)),
        ColorSequenceKeypoint.new(1, Color3.fromRGB(255, 105, 180))
    }
    BtnGradient.Parent = Button

    -- Green Dot and Status
    local StatusFrame = Instance.new("Frame")
    StatusFrame.Size = UDim2.new(1, 0, 0, 10)
    StatusFrame.Position = UDim2.new(0, 0, 0, 40)
    StatusFrame.BackgroundTransparency = 1
    StatusFrame.Parent = ButtonFrame

    local StatusDot = Instance.new("Frame")
    StatusDot.Size = UDim2.new(0, 8, 0, 8)
    StatusDot.Position = UDim2.new(0, 5, 0, 1)
    StatusDot.BackgroundColor3 = Color3.fromRGB(0, 255, 0)
    StatusDot.Parent = StatusFrame
    local DotCorner = Instance.new("UICorner")
    DotCorner.CornerRadius = UDim.new(1, 0)
    DotCorner.Parent = StatusDot

    local StatusLabel = Instance.new("TextLabel")
    StatusLabel.Size = UDim2.new(0, 100, 0, 10)
    StatusLabel.Position = UDim2.new(0, 15, 0, 0)
    StatusLabel.BackgroundTransparency = 1
    StatusLabel.Text = "Undetected and Working"
    StatusLabel.TextColor3 = Color3.fromRGB(0, 255, 0)
    StatusLabel.TextSize = 12
    StatusLabel.Font = Enum.Font.Gotham
    StatusLabel.TextXAlignment = Enum.TextXAlignment.Left
    StatusLabel.Parent = StatusFrame

    Button.MouseButton1Click:Connect(function()
        local key = KeyInput.Text
        if key == "" or key == "Enter your key here" then
            game:GetService("StarterGui"):SetCore("SendNotification", {
                Title = "Error",
                Text = "Please enter a valid key",
                Duration = 3
            })
            return
        end
        -- Execute script with enhanced error handling
        local success, result = pcall(function()
            local scriptContent = game:HttpGet(script.url, true)
            if not scriptContent or scriptContent == "" then
                error("Failed to fetch script content")
            end
            local scriptCode = string.format("script_key='%s';%s", key, scriptContent)
            local func = loadstring(scriptCode)
            if not func then
                error("Failed to compile script")
            end
            return func()
        end)
        if not success then
            game:GetService("StarterGui"):SetCore("SendNotification", {
                Title = "Execution Error",
                Text = "Failed to execute " .. script.name .. ": " .. tostring(result),
                Duration = 5
            })
        else
            game:GetService("StarterGui"):SetCore("SendNotification", {
                Title = "Success",
                Text = script.name .. " executed successfully!",
                Duration = 3
            })
        end
    end)
end

-- Minimize functionality
local isMinimized = false
local function toggleMinimize()
    if isMinimized then
        -- Restore GUI
        MainFrame.Visible = true
        local restoreTween = TweenService:Create(MainFrame, TweenInfo.new(0.3, Enum.EasingStyle.Back, Enum.EasingDirection.Out), {Size = UDim2.new(0, 300, 0, 400)})
        restoreTween:Play()
        if isMobile then
            MinimizeButton.Text = "-"
        end
        isMinimized = false
    else
        -- Minimize GUI
        local minimizeTween = TweenService:Create(MainFrame, TweenInfo.new(0.3, Enum.EasingStyle.Back, Enum.EasingDirection.In), {Size = UDim2.new(0, 40, 0, 40)})
        minimizeTween:Play()
        minimizeTween.Completed:Connect(function()
            MainFrame.Visible = isMobile -- Keep visible on mobile for minimize button
        end)
        if isMobile then
            MinimizeButton.Text = "+"
        end
        isMinimized = true
    end
end

-- Right Ctrl to minimize (PC)
UserInputService.InputBegan:Connect(function(input, gameProcessedEvent)
    if not gameProcessedEvent and input.KeyCode == Enum.KeyCode.RightControl then
        toggleMinimize()
    end
end)

-- Minimize Button (Mobile)
if isMobile then
    MinimizeButton.MouseButton1Click:Connect(toggleMinimize)
end

-- Startup Animation
local spinTween = TweenService:Create(StarImage, TweenInfo.new(2, Enum.EasingStyle.Quad, Enum.EasingDirection.InOut, 0, false), {Rotation = 360})
local stretchTween = TweenService:Create(StarImage, TweenInfo.new(1, Enum.EasingStyle.Quad, Enum.EasingDirection.InOut), {Size = UDim2.new(0, 200, 0, 200)})
spinTween:Play()
stretchTween:Play()
stretchTween.Completed:Connect(function()
    -- Fade out startup screen
    local fadeTween = TweenService:Create(StartupFrame, TweenInfo.new(0.5, Enum.EasingStyle.Linear), {BackgroundTransparency = 1})
    local imageFadeTween = TweenService:Create(StarImage, TweenInfo.new(0.5, Enum.EasingStyle.Linear), {ImageTransparency = 1})
    fadeTween:Play()
    imageFadeTween:Play()
    imageFadeTween.Completed:Connect(function()
        StartupFrame:Destroy()
        -- Ensure MainFrame is visible and stretch it
        MainFrame.Visible = true
        local menuStretch = TweenService:Create(MainFrame, TweenInfo.new(0.5, Enum.EasingStyle.Back, Enum.EasingDirection.Out), {Size = UDim2.new(0, 300, 0, 400)})
        menuStretch:Play()
    end)
end)

-- Clear placeholder text on focus
KeyInput.Focused:Connect(function()
    if KeyInput.Text == "Enter your key here" then
        KeyInput.Text = ""
    end
end)

-- Reset placeholder text if empty
KeyInput.FocusLost:Connect(function()
    if KeyInput.Text == "" then
        KeyInput.Text = "Enter your key here"
    end
end)

-- Make GUI draggable
local dragging
local dragInput
local dragStart
local startPos

MainFrame.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        dragging = true
        dragStart = input.Position
        startPos = MainFrame.Position
        input.Changed:Connect(function()
            if input.UserInputState == Enum.UserInputState.End then
                dragging = false
            end
        end)
    end
end)

MainFrame.InputChanged:Connect(function(input)
    if (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) and dragging then
        local delta = input.Position - dragStart
        MainFrame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
    end
end)
